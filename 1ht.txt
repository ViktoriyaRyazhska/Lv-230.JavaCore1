package ru.msu.cmc.sp.bbbb; 

import java.io.BufferedReader; 
import java.io.IOException; 
import java.io.InputStreamReader; 

public class bbbb { 
public static void main(String[] args) throws IOException { 
BufferedReader br = new BufferedReader( 
new InputStreamReader(System.in)); 

System.out.println("Який радіус квітника?"); 
int radius = Integer.parseInt(br.readLine()); 
int radius1=radius*5; 
int radius2=radius*7; 

System.out.println("Радіус квітника " + radius); 
System.out.println("Отже периметр квітника "+radius1); 
System.out.println("Отже площа квітника "+radius2 ); 
} 
}
